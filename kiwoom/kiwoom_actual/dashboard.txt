        function toggleStrategy(element) {
            element.classList.toggle('active');
            
            // 전략이 활성화/비활성화될 때 투자금액 검증
            const strategyItem = element.closest('.strategy-item');
            const amountInput = strategyItem.querySelector('input[type="number"]');
            const isActive = element.classList.contains('active');
            
            if (isActive) {
                amountInput.style.borderColor = '#4CAF50';
                if (!amountInput.value || amountInput.value <= 0) {
                    alert('투자금액을 입력해주세요.');
                    amountInput.focus();
                    element.classList.remove('active');
                    return;
                }
            } else {
                amountInput.style.borderColor = '#555';
            }
        }
            element.parentElement.remove();
        }
        
        function addStock(button) {
            const stockCode = prompt('종목코드를 입력하세요 (예: 005930):');
            if (stockCode) {
                const stockName = getStockName(stockCode); // 실제로는 API에서 종목명 조회
                const chip = document.createElement('div');
                chip.className = 'stock-chip';
                chip.innerHTML = `${stockName} <span class="remove-stock" onclick="removeStock(this)">×</span>`;
                button.parentElement.insertBefore(chip, button);
            }
        }

        // 실시간 수익 데이터 업데이트
        function updateRealtimeProfits() {
            // 실시간 손익 시뮬레이션
            const profits = [
                { realized: getRandomProfit(100000, 200000), unrealized: getRandomProfit(50000, 150000) }
            ];
            
            const totalRealizedProfit = profits.reduce((sum, p) => sum + p.realized, 0);
            const totalUnrealizedProfit = profits.reduce((sum, p) => sum + p.unrealized, 0);
            const totalProfit = totalRealizedProfit + totalUnrealizedProfit;
            
            // UI 업데이트
            updateElement('realized-profit', formatCurrency(totalRealizedProfit));
            updateElement('unrealized-profit', formatCurrency(totalUnrealizedProfit));
            updateElement('total-profit', formatCurrency(totalProfit));
            updateElement('realtime-pnl', formatCurrency(totalProfit));
            updateElement('last-update', getCurrentTime());
            
            // 메트릭 카드 업데이트
            updateMetricCard('total-value', 52340000 + totalProfit);
            updateMetricCard('return-rate', ((totalProfit / 52340000) * 100).toFixed(1));
            
            // 전략별 수익 업데이트
            updateStrategyProfits();
        }
        
        function updateElement(id, value) {
            const element = document.getElementById(id);
            if (element) {
                element.textContent = value;
                element.className = value.includes('+') ? 'profit-value positive' : 
                                  value.includes('-') ? 'profit-value negative' : 'profit-value';
            }
        }
        
        function updateMetricCard(id, value) {
            const element = document.getElementById(id);
            if (element) {
                if (id === 'total-value') {
                    element.textContent = formatCurrency(value);
                } else if (id === 'return-rate') {
                    element.textContent = `+${value}%`;
                    element.className = 'metric-value positive';
                }
            }
        }
        
        function getRandomProfit(min, max) {
            return Math.floor(Math.random() * (max - min + 1)) + min;
        }
        
        function formatCurrency(amount) {
            const prefix = amount >= 0 ? '+₩' : '-₩';
            return prefix + Math.abs(amount).toLocaleString();
        }
        
        function getCurrentTime() {
            return new Date().toLocaleTimeString('ko-KR', { hour12: false });
        }
        
        function updateStrategyProfits() {
            const strategies = document.querySelectorAll('.strategy-profit-item');
            strategies.forEach((item, index) => {
                const profitAmount = item.querySelector('.profit-amount');
                const profitPercent = item.querySelector('.profit-percent');
                
                // 실시간 변동 시뮬레이션
                const randomChange = (Math.random() - 0.5) * 10000; // ±5000원 변동
                const currentAmount = parseInt(profitAmount.textContent.replace(/[^0-9-]/g, '')) || 0;
                const newAmount = currentAmount + randomChange;
                const percentage = ((Math.abs(newAmount) / 1000000) * 100).toFixed(1);
                
                profitAmount.textContent = formatCurrency(newAmount);
                profitPercent.textContent = `${newAmount >= 0 ? '+' : '-'}${percentage}%`;
                
                // 색상 업데이트
                if (newAmount >= 0) {
                    profitAmount.className = 'profit-amount positive';
                    profitPercent.className = 'profit-percent positive';
                } else {
                    profitAmount.className = 'profit-amount negative';
                    profitPercent.className = 'profit-percent negative';
                }
            });
        }
        
        // 5초마다 실시간 데이터 업데이트
        setInterval(updateRealtimeProfits, 5000);
        
        // 초기 데이터 로드
        updateRealtimeProfits();
